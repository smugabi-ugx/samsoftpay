response = requests.post(
    "{{ request.host_url }}v1/charges",

        "Authorization": "Bearer {{ current_user.secret_key }}",

        "X-Timestamp": str(int(time.time())),
        "Content-Type": "application/json",
    },
    json={
        "amount": 5000,
        "currency": "UGX",
        "channel": "mtn_momo",
        "customer": {"phone": "+256700123456"},
        "reference": "order-001",
    }
)
